load playerTexture from "playerTexture.png"

const playerSpeed: int = 1

func main(){

    playerSprite: Object ptr = getPlayerSprite(10, 120, playerTexture[0])

    playing: bool = true
    while playing{
        input: PlayerInput = pollPlayerInput()

        if input.right{
            @playerSprite.x += playerSpeed
        }
        else if input.x{
            playing = false
        }

        waitOnFrame()
    }

}
